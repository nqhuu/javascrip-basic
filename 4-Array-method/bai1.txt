// Bài 1:  
// a. Thêm số 8 vào cuối array data2 . 
var data2 = [1,1,3,4,4,5,6,7,9]
data2.push(8)
console.log(data2);
// b.  Thêm số 10 vào cuối array data2
data2.push(10)
console.log(data2);
// c. Thêm số 5 vào đầu array data2 .
data2.unshift(5)
console.log(data2);
// d. Đổi phần tử có vị trí là 5 thành số 11 (nếu dùng index ko tồn tại trong mảng thì nó sẽ thành thêm chứ ko phải thay thế)
data2[5] = 11;
console.log(data2);
// e. Xóa phần tử vị trí  thứ 4 và thêm hai số 5 và 7 vào data2 - 
// sdung array.splice(startIndex, deleteCount, items).
// startIndex: Vị trí bắt đầu để thay đổi mảng. 
// deleteCount: tổng số phần tử muốn xóa.
// items: phần tử thêm vào mảng.
data2.splice(4,1,5,7);
console.log(data2);
